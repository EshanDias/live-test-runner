## Context & Constraints

- Jest JSON gives us console output at **file level only** via the `console` array on each file result.
- Suite and test level outputs will be **empty by default** and that is correct — do not back-fill them.
- When a file runs, only `FileResult.output` is written. Suite and test outputs are never touched by a file run.
- In the future, individual test reruns will write to `TestResult.output`. The storage shape must support this now without any structural changes later.
- All scoping logic lives in the **extension**, not the webview. The webview only renders what it receives.

### 2.2 Add `buildScopedLogPayload()` helper

Add this function in `extension.ts`. It is called whenever a selection changes or a run completes
with an active selection.

```ts
type LogSection = {
  label:      string
  scope:      'file' | 'suite' | 'test'
  capturedAt: number | null
  lines:      OutputLine[]
}

type ErrorEntry = {
  testName:        string
  failureMessages: string[]
  capturedAt:      number | null
}

type ErrorSection = {
  label:    string
  scope:    'file' | 'suite' | 'test'
  errors:   ErrorEntry[]
}

type ScopedLogPayload = {
  logSections:   LogSection[]
  errorSections: ErrorSection[]
}

function buildScopedLogPayload(scope: SelectionScope): ScopedLogPayload {
  const { level, fileId, suiteId, testId } = scope

  if (level === 'file') {
    return buildFilePayload(fileId)
  }
  if (level === 'suite') {
    return buildSuitePayload(fileId, suiteId!)
  }
  return buildTestPayload(fileId, suiteId!, testId!)
}
```

### 2.3 `buildFilePayload(fileId)`

```ts
function buildFilePayload(fileId: string): ScopedLogPayload {
  const fileOutput  = resultStore.getFileOutput(fileId)
  const fileResult  = resultStore.getFile(fileId)

  // --- Log sections ---
  const logSections: LogSection[] = []

  // File section — always included
  logSections.push({
    label:      fileResult.filePath,
    scope:      'file',
    capturedAt: fileOutput.capturedAt,
    lines:      fileOutput.lines,
  })

  // Suite sections — only if capturedAt is set AND has lines
  for (const [suiteId, suite] of fileResult.suites) {
    const suiteOutput = resultStore.getSuiteOutput(fileId, suiteId)
    if (suiteOutput.capturedAt !== null && suiteOutput.lines.length > 0) {
      logSections.push({
        label:      suite.name,
        scope:      'suite',
        capturedAt: suiteOutput.capturedAt,
        lines:      suiteOutput.lines,
      })
    }

    // Test sections — only if capturedAt is set AND has lines
    for (const [testId, test] of suite.tests) {
      const testOutput = resultStore.getTestOutput(fileId, suiteId, testId)
      if (testOutput.capturedAt !== null && testOutput.lines.length > 0) {
        logSections.push({
          label:      test.name,
          scope:      'test',
          capturedAt: testOutput.capturedAt,
          lines:      testOutput.lines,
        })
      }
    }
  }

  // --- Error sections ---
  const errorSections: ErrorSection[] = []
  const fileErrors: ErrorEntry[] = []

  for (const [suiteId, suite] of fileResult.suites) {
    for (const [testId, test] of suite.tests) {
      if (test.failureMessages.length > 0) {
        fileErrors.push({
          testName:        test.name,
          failureMessages: test.failureMessages,
          capturedAt:      resultStore.getTestOutput(fileId, suiteId, testId).capturedAt
                           ?? fileOutput.capturedAt,
        })
      }
    }
  }

  if (fileErrors.length > 0) {
    errorSections.push({
      label:  fileResult.filePath,
      scope:  'file',
      errors: fileErrors,
    })
  }

  return { logSections, errorSections }
}
```

### 2.4 `buildSuitePayload(fileId, suiteId)`

```ts
function buildSuitePayload(fileId: string, suiteId: string): ScopedLogPayload {
  const suite       = resultStore.getSuite(fileId, suiteId)
  const suiteOutput = resultStore.getSuiteOutput(fileId, suiteId)

  // --- Log sections ---
  const logSections: LogSection[] = []

  // Suite section — always included (user explicitly selected it)
  logSections.push({
    label:      suite.name,
    scope:      'suite',
    capturedAt: suiteOutput.capturedAt,
    lines:      suiteOutput.lines,
  })

  // Test sections within this suite — only if capturedAt set AND has lines
  for (const [testId, test] of suite.tests) {
    const testOutput = resultStore.getTestOutput(fileId, suiteId, testId)
    if (testOutput.capturedAt !== null && testOutput.lines.length > 0) {
      logSections.push({
        label:      test.name,
        scope:      'test',
        capturedAt: testOutput.capturedAt,
        lines:      testOutput.lines,
      })
    }
  }

  // --- Error sections ---
  const suiteErrors: ErrorEntry[] = []

  for (const [testId, test] of suite.tests) {
    if (test.failureMessages.length > 0) {
      suiteErrors.push({
        testName:        test.name,
        failureMessages: test.failureMessages,
        capturedAt:      resultStore.getTestOutput(fileId, suiteId, testId).capturedAt,
      })
    }
  }

  const errorSections: ErrorSection[] = suiteErrors.length > 0
    ? [{ label: suite.name, scope: 'suite', errors: suiteErrors }]
    : []

  return { logSections, errorSections }
}
```

### 2.5 `buildTestPayload(fileId, suiteId, testId)`

```ts
function buildTestPayload(fileId: string, suiteId: string, testId: string): ScopedLogPayload {
  const test       = resultStore.getTest(fileId, suiteId, testId)
  const testOutput = resultStore.getTestOutput(fileId, suiteId, testId)

  // Log section — always included even if empty (user explicitly selected this test)
  const logSections: LogSection[] = [{
    label:      test.name,
    scope:      'test',
    capturedAt: testOutput.capturedAt,
    lines:      testOutput.lines,
  }]

  // Error section — only if failed
  const errorSections: ErrorSection[] = test.failureMessages.length > 0
    ? [{
        label:  test.name,
        scope:  'test',
        errors: [{
          testName:        test.name,
          failureMessages: test.failureMessages,
          capturedAt:      testOutput.capturedAt,
        }],
      }]
    : []

  return { logSections, errorSections }
}
```

### 5.3 `renderScopedOutput(payload)` — Column 2

```js
function renderScopedOutput(payload) {
  renderLogSections(payload.logSections)
  renderErrorSections(payload.errorSections)
}

function renderLogSections(sections) {
  const container = document.getElementById('col2-content')
  container.innerHTML = ''

  if (sections.length === 0) {
    container.innerHTML = '<p class="empty-state">No output captured yet. Run a test to see logs here.</p>'
    return
  }

  // Check if the ONLY section present is the selected scope with no lines
  // (e.g. user clicked a test that has never been run individually)
  const allEmpty = sections.every(s => s.lines.length === 0)
  if (allEmpty) {
    const scope = sections[0].scope
    const msg = scope === 'test'
      ? 'No output captured at test level. Check file level for logs — these are the closest available.'
      : scope === 'suite'
      ? 'No output captured at suite level. Check file level for logs.'
      : 'No output captured yet. Run a test to see logs here.'
    container.innerHTML = `<p class="empty-state">${msg}</p>`
    return
  }

  for (const section of sections) {
    if (section.lines.length === 0) continue   // skip empty sections silently

    const sectionEl = document.createElement('div')
    sectionEl.className = 'output-section'
    sectionEl.dataset.scope = section.scope

    const header = document.createElement('div')
    header.className = 'section-header'
    header.innerHTML = `
      <span class="scope-icon">${scopeIcon(section.scope)}</span>
      <span class="scope-label">${escapeHtml(section.label)}</span>
      <span class="captured-at">${formatCapturedAt(section.capturedAt)}</span>
    `
    sectionEl.appendChild(header)

    const lines = document.createElement('div')
    lines.className = 'section-lines'
    for (const line of section.lines) {
      const lineEl = document.createElement('div')
      lineEl.className = `log-line log-line-${line.level}`
      lineEl.dataset.level = line.level
      lineEl.textContent = line.text
      lines.appendChild(lineEl)
    }
    sectionEl.appendChild(lines)
    container.appendChild(sectionEl)
  }

  // Apply active tab filter after render
  applyTabFilter(activeTab)
}
```

### 5.4 `renderErrorSections(sections)` — Column 3

```js
function renderErrorSections(sections) {
  const container = document.getElementById('col3-content')
  container.innerHTML = ''

  if (sections.length === 0 || sections.every(s => s.errors.length === 0)) {
    container.innerHTML = '<p class="empty-state no-errors">No failures for this selection. ✓</p>'
    return
  }

  for (const section of sections) {
    for (const entry of section.errors) {
      const entryEl = document.createElement('div')
      entryEl.className = 'error-entry'

      const nameEl = document.createElement('div')
      nameEl.className = 'error-test-name'
      nameEl.textContent = entry.testName
      entryEl.appendChild(nameEl)

      for (const msg of entry.failureMessages) {
        const msgEl = document.createElement('pre')
        msgEl.className = 'error-message'
        msgEl.textContent = msg   // ANSI already stripped upstream
        entryEl.appendChild(msgEl)
      }

      container.appendChild(entryEl)
    }
  }
}
```

### 5.5 Tab filtering

Tabs (Logs / Info / Warn / Error) filter **all sections simultaneously**. A section header is
hidden if it has no lines matching the active tab.

```js
let activeTab = 'log'

function applyTabFilter(level) {
  activeTab = level

  // Show/hide individual lines
  document.querySelectorAll('.log-line').forEach(el => {
    el.style.display = (level === 'all' || el.dataset.level === level) ? '' : 'none'
  })

  // Hide section entirely if it has no visible lines
  document.querySelectorAll('.output-section').forEach(section => {
    const hasVisible = [...section.querySelectorAll('.log-line')]
      .some(el => el.style.display !== 'none')
    section.style.display = hasVisible ? '' : 'none'
  })
}
```

Tab buttons call `applyTabFilter('log')`, `applyTabFilter('warn')` etc. Add an "All" tab option
that passes `'all'` to show everything.

### 5.6 Helper functions

```js
function scopeIcon(scope) {
  return { file: '📄', suite: '🔷', test: '🔹' }[scope] ?? '•'
}

function formatCapturedAt(capturedAt) {
  if (capturedAt === null) return ''
  const diff = Date.now() - capturedAt
  if (diff < 10_000)  return 'captured just now'
  if (diff < 120_000) return `captured ${Math.floor(diff / 1000)}s ago`
  return `captured ${new Date(capturedAt).toLocaleTimeString()}`
}

function escapeHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
}
```

## Behaviour Summary

| User action | Column 2 shows | Column 3 shows |
|-------------|---------------|----------------|
| Clicks file | File section (always). Suite sections only if individually run with output. Test sections only if individually run with output OR has failures. | All failed tests in the file. |
| Clicks suite | Suite section (always, even if empty — show placeholder). Test sections only if individually run with output. | All failed tests in the suite. |
| Clicks test | Test section (always, even if empty — show placeholder). | That test's failure only, or ✓ empty state. |
| Full file run completes, file is selected | Column 2 refreshes automatically with updated file section timestamp. | Column 3 refreshes with current failures. |
| Single test reruns (future) | Test section appears / updates with new timestamp alongside existing file section. | Test section in Column 3 updates. |
